const {spec} = finch('data/iris.csv')
  .width(150)
  .height(150)
  .bar({binSpacing: 0})
  .x('sepalLength', 'q', {
    bin: true
  })
  .y(null, 'q', {aggregate: 'count'})
  .column('species', 'n');
